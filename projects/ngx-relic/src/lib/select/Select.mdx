## Overview

Import `SelectModule` to your module before using `<ngx-select>` component

```ts
import { SelectModule } from "ngx-relic";

@NgModule({
  declarations: [AppComponent],
  imports: [BrowserModule, SelectModule],
  bootstrap: [AppComponent],
})
export class AppModule {}
```
