## Overview

Import `DatepickerModule` to your module before using `<ngx-datepicker>` component

```ts
import { DatepickerModule } from "ngx-relic";

@NgModule({
  declarations: [AppComponent],
  imports: [BrowserModule, DatepickerModule],
  bootstrap: [AppComponent],
})
export class AppModule {}
```
