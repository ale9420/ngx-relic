## Overview

Import `LabelModule` to your module before using `<ngx-label>` component

```ts
import { LabelModule } from "ngx-relic";

@NgModule({
  declarations: [AppComponent],
  imports: [BrowserModule, LabelModule],
  bootstrap: [AppComponent],
})
export class AppModule {}
```
