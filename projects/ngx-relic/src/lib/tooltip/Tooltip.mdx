## Overview

Import `TooltipModule` to your module before using `<ngx-tooltip>` component

```ts
import { TooltipModule } from "ngx-relic";

@NgModule({
  declarations: [AppComponent],
  imports: [BrowserModule, TooltipModule],
  bootstrap: [AppComponent],
})
export class AppModule {}
```
