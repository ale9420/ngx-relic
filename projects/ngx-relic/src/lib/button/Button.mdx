## Overview

Import `ButtonModule` to your module before using `<ngx-button>` component

```ts
import { ButtonModule } from "ngx-relic";

@NgModule({
  declarations: [AppComponent],
  imports: [BrowserModule, ButtonModule],
  bootstrap: [AppComponent],
})
export class AppModule {}
```
