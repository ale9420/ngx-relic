## Overview

Import `InputModule` to your module before using `<ngx-input>` component

```ts
import { InputModule } from "ngx-relic";

@NgModule({
  declarations: [AppComponent],
  imports: [BrowserModule, InputModule],
  bootstrap: [AppComponent],
})
export class AppModule {}
```
