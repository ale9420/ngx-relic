## Overview

Import `TagModule` to your module before using `<ngx-tag>` component

```ts
import { TagModule } from "ngx-relic";

@NgModule({
  declarations: [AppComponent],
  imports: [BrowserModule, TagModule],
  bootstrap: [AppComponent],
})
export class AppModule {}
```
